% Dokumentenkopf ---------------------------------------------------------------------------------------
%   Diese Vorlage basiert auf "scrreprt" aus dem koma-script.
% ------------------------------------------------------------------------------------------------------
\documentclass[
    12pt, % Schriftgröße
    DIV=10, % Änderung der Größe des Satzspiegels (bedruckbarer Bereich einer Seite), nur in Verbindung mit koma-script verwendbar
    ngerman, % für Umlaute, Silbentrennung etc.
    a4paper, % Papierformat
    oneside, % einseitiges Dokument
    titlepage, % es wird eine Titelseite verwendet
    parskip=half, % Abstand zwischen Absätzen (halbe Zeile)
    headings=normal, % Größe der Überschriften verkleinern
    listof=totoc, % Verzeichnisse im Inhaltsverzeichnis aufführen
    bibliography=totoc, % Literaturverzeichnis im Inhaltsverzeichnis aufführen
    index=totoc, % Index im Inhaltsverzeichnis aufführen
    captions=tableheading, % Beschriftung von Tabellen unterhalb ausgeben
    final % Status des Dokuments (final/draft)
]{scrreprt}

% UTF8 und T1 Fontencoding -----------------------------------------------------------------------------
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}
\usepackage{tikz} % Ensure TikZ/PGF is loaded after fontenc/inputenc
\usepackage{array}


\input{Preamble/Meta}
\input{Preamble/Packages}
\input{Preamble/Seitenstil}

\begin{document}
\include{Preamble/Silbentrennung}
\include{Preamble/Befehle}


\setcounter{secnumdepth}{3}

% Deckblatt und Abstract ohne Seitenzahl ---------------------------------------------------------------
\include{Formalien/Deckblatt}

% Seitennummerierung -----------------------------------------------------------------------------------
%   Vor dem Hauptteil werden die Seiten in großen römischen Ziffern 
%   nummeriert, beginnend mit II
% ------------------------------------------------------------------------------------------------------
\pagenumbering{Roman}
\setcounter{page}{2}  % Start with Roman numeral II

% Hauptinhaltsverzeichnis - nur bis Chapter-Level
\tableofcontents{}

% Abkürzungsverzeichnis --------------------------------------------------------------------------------
\input{Inhalt/Glossar}

% arabische Seitenzahlen im Hauptteil ------------------------------------------------------------------
\clearpage{}
\pagenumbering{arabic}
\pagestyle{scrheadings}

% die Inhaltskapitel werden in "Inhalt.tex" inkludiert -------------------------------------------------
\input{Preamble/Inhalt}

% Literaturverzeichnis ---------------------------------------------------------------------------------
\chapter{Literaturverzeichnis}
\printbibliography[heading=none]

% Restliche Verzeichnisse -> wenn diese leer sind, sollte das hier auskommentiert werden ------------------------------------------------------------------------------
% Abbildungsverzeichnis
\cleardoublepage
\chapter{Abbildungsverzeichnis}
\makeatletter
\@starttoc{lof}  % Verwendet direkt die .lof-Datei ohne Überschrift zu erzeugen
\makeatother

% Ehrenwörtliche Erklärung ----------------------------------------------------------------------------
\cleardoublepage
\include{Formalien/Erklaerung} 

\chapter{Anhang}
\section*{Anhangsverzeichnis}
\makeatletter
\@starttoc{app} % liest Einträge, die wir mit \addcontentsline{app}{...}{...} geschrieben haben
\makeatother

\input{Inhalt/Anhang}

\end{document}
